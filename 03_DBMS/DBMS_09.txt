PROCEDURE 하는 방법

DELIMITER //
CREATE PROCEDURE 이름(IN 000 INT, ...)
BEGIN 쿼리문; END: //
DELIMITER ;

(OUT을 사용하게 되면 잘 불러내야한다)
= CALL 프로시저이름(@000);
SELECT @000 FROM 테이블;
